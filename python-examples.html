# CLASS VARIABLES

class Cars:
    def __init__(self, model, year, price):
        self.model = model
        self.year = year
        self.price = price
    
    def car_details(self):
        return '{} {} {}'.format(self.model, self.year, self.price)

    def price_inc(self):
        self.price = int(self.price*1.15)

c1 = Cars('Honda','2018','100000')
c2 = Cars('Hyundai','2017','80000')

print(c1.car_details())
print(c2.car_details())

print(c1.model())
print(c1.price_inc())
print(c1.price())

class Student:

    def __init__(self, s1, s2, s3):
        self.s1 = s1
        self.s2 = s2
        self.s3 = s3
    
    def avg(self):
        return (self.s1 + self.s2 + self.s3)/ 3

n1 = Student(23, 45, 67)
n2 = Student(28, 89, 45)

print(n1.avg())

class Employee:

    pay_hike = 1.04

    def __init__(self, first, last, pay):
        self.first = first
        self.last = last
        self.pay = pay

    def details(self):
        return '{} {} {}'.format(self.first, self.last, self.pay)

    def increase(self):
        self.pay = int(self.pay * self.pay_hike)

e1 = Employee('Krishna','Mohan',50000)
e2 = Employee('Krishna','Mohan',60000)

print(e1.details())
e1.increase()
print(e1.details())
